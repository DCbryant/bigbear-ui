## 可操作

```tsx
export function ProgressExample(){
    const [count,setCount]=useState(30)
    return (
        <div>
            <Button onClick={()=>setCount(count+1)}>+1</Button>
            <Button onClick={()=>setCount(count-1)}>-1</Button>
            <Button onClick={()=>setCount(count+10)}>+10</Button>
            <Button onClick={()=>setCount(count-10)}>-10</Button>
            <Progress count={count}></Progress>
        </div>
    )
}
```
